# Provisioning

```
INFLUX_CMD="influx -ssl -unsafeSsl -username ${INFLUXDB_ADMIN_USER} -password ${INFLUXDB_ADMIN_PASSWORD} -execute"

$INFLUX_CMD "CREATE DATABASE \"sondas\" WITH DURATION 365d REPLICATION 1 SHARD DURATION 24h NAME \"autogen\";"
$INFLUX_CMD "CREATE USER ${INFLUXDB_SONDAS_USER} WITH PASSWORD "\'"${INFLUXDB_SONDAS_PASSWORD}"\'";"
$INFLUX_CMD "GRANT ALL ON \"${INFLUXDB_SONDAS_USER}\" TO \"sondas\";"

$INFLUX_CMD "CREATE DATABASE \"telegraf\" WITH DURATION 365d REPLICATION 1 SHARD DURATION 24h NAME \"autogen\";"
$INFLUX_CMD "CREATE USER ${INFLUXDB_TELEGRAF_USER} WITH PASSWORD "\'"${INFLUXDB_TELEGRAF_PASSWORD}"\'";"
$INFLUX_CMD "GRANT ALL ON \"${INFLUXDB_TELEGRAF_USER}\" TO \"telegraf\";"

$INFLUX_CMD "CREATE DATABASE \"${INFLUXDB_SLATOR_DATABASE}\" WITH DURATION 0d REPLICATION 1 SHARD DURATION 14d NAME \"autogen\";"
$INFLUX_CMD "CREATE USER ${INFLUXDB_SLATOR_USER} WITH PASSWORD "\'"${INFLUXDB_SLATOR_PASSWORD}"\'";"
$INFLUX_CMD "GRANT ALL ON \"${INFLUXDB_SLATOR_DATABASE}\" TO \"${INFLUXDB_SLATOR_USER}\";"

$INFLUX_CMD "CREATE DATABASE \"${INFLUXDB_SYNTHETIX_MANAGER_DATABASE}\" WITH DURATION 365d REPLICATION 1 SHARD DURATION 24h NAME \"autogen\";"
$INFLUX_CMD "CREATE USER ${INFLUXDB_SYNTHETIX_MANAGER_USER} WITH PASSWORD "\'"${INFLUXDB_SYNTHETIX_MANAGER_PASSWORD}"\'";"
$INFLUX_CMD "GRANT ALL ON \"${INFLUXDB_SYNTHETIX_MANAGER_DATABASE}\" TO \"${INFLUXDB_SYNTHETIX_MANAGER_USER}\";"

$INFLUX_CMD "CREATE DATABASE \"${INFLUXDB_SYNTHETIX_UI_DATABASE}\" WITH DURATION 365d REPLICATION 1 SHARD DURATION 24h NAME \"autogen\";"
$INFLUX_CMD "CREATE USER ${INFLUXDB_SYNTHETIX_UI_USER} WITH PASSWORD "\'"${INFLUXDB_SYNTHETIX_UI_PASSWORD}"\'";"
$INFLUX_CMD "GRANT ALL ON \"${INFLUXDB_SYNTHETIX_UI_DATABASE}\" TO \"${INFLUXDB_SYNTHETIX_UI_USER}\";"
```
